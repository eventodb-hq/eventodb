 ### API Examples                                                               
                                                                                
 Create Namespace:                                                              
                                                                                
 ```json                                                                        
   Request:  ["ns.create", "tenant-a", {"description": "Production tenant"}]    
   Response: {                                                                  
     "namespace": "tenant-a",                                                   
     "token": "ns_dGVuYW50LWE_a7f3c8d9...",                                     
     "createdAt": "2024-12-17T06:44:20.123Z"                                    
   }                                                                            
 ```                                                                            
                                                                                
 Delete Namespace:                                                              
                                                                                
 ```json                                                                        
   Request:  ["ns.delete", "tenant-a"]                                          
   Response: {                                                                  
     "namespace": "tenant-a",                                                   
     "deletedAt": "2024-12-17T06:45:00.123Z",                                   
     "messagesDeleted": 0                                                       
   }                                                                            
 ```                                                                            
                                                                                
 List Namespaces:                                                               
                                                                                
 ```json                                                                        
   Request:  ["ns.list"]                                                        
   Response: [                                                                  
     {                                                                          
       "namespace": "default",                                                  
       "description": "Default namespace",                                      
       "createdAt": "2024-12-17T06:00:00.123Z",                                 
       "messageCount": 0                                                        
     }                                                                          
   ]                                                                            
 ```                                                                            
                                                                                
 Get Namespace Info:                                                            
                                                                                
 ```json                                                                        
   Request:  ["ns.info", "tenant-a"]                                            
   Response: {                                                                  
     "namespace": "tenant-a",                                                   
     "description": "Production tenant",                                        
     "createdAt": "2024-12-17T06:44:20.123Z",                                   
     "messageCount": 0,                                                         
     "streamCount": 0,                                                          
     "lastActivity": null                                                       
   }                                                                            
 ```
